Module Sundials.Roots


module Roots: sig .. end
Utility functions for arrays of roots (zero-crossings).
type t = Sundials.int_array 
type val_array = Sundials.Carray.t

type root_event =
| NoRoot (*No root (0)*)
| Rising (*Rising root (1)*)
| Falling (*Falling root (-1)*) 
val empty : t
An array with 0 elements.
val create : int -> t
create n returns an array with n elements.
val length : t -> int
Returns the length of an array
val print : t -> unit
print r prints a line containing a tab-delimited list of the values of r (in the format "% d", where 0 = NoRoot, 1 = Rising, -1 = Falling), and then a newline.
val get : t -> int -> bool
get r i returns true if the value of the ith element of r is either Rising or Falling.
val get' : t -> int -> root_event
get r i returns the value of the ith element of r.
val set : t -> int -> root_event -> unit
set r i v sets the value of the ith element of r.
val to_int : root_event -> int
Returns 0 for NoRoot, 1 for Rising, and -1 for Falling.
val from_int : int -> root_event
Returns NoRoot for 0, Rising for 1, and Falling for -1.
val reset : t -> unit
Resets all elements to NoRoot.
val exists : t -> bool
Returns true if any elements are equal to Rising or Falling.
val appi : (int -> root_event -> unit) -> t -> unit
appi f r applies f to the indexes and values of each element in r.
